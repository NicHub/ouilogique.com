#!/bin/bash
filelist=`ls -1 *.jpeg | sort -n -k2 | while read -r i; do printf "'$i' "; done`
fileout=`date "+%Y-%m-%d_%H-%M-%S_out.pdf"`
eval "gm convert -quality 10 -level 20%,1,85% $filelist $fileout"







# ImageMagick.org
# http://www.imagemagick.org/Usage/formats/#pdf
file=`date "+%Y-%m-%d_%H-%M-%S_out.pdf"`; convert *.tiff -monochrome -compress Group4 $file; open $file
convert *.tiff *.tif *.png *.jpg *.jpeg  -alpha off -monochrome -compress Group4 -quality 100 -units PixelsPerInch -density 300  `date "+%Y-%m-%d_%H-%M-%S_out.pdf"`
convert -density 600 in.pdf `date "+%Y-%m-%d_%H-%M-%S_out.png"`

# Convert to jpeg quality 70 in sibling folder "out"
for pic in *.jpg ; do echo "$pic" ; convert "$pic" -quality 70 "../out/${pic%.jpg}.jpeg" ; done

# Créer des thumbnails
mkdir ../thumbs; for pic in *.* ; do echo "$pic" ; convert "$pic" -thumbnail 'x150' "../thumbs/${pic}" ; done

# Créer des thumbnails de même taille pour Wordpress Gallery (le flag -define sert à rendre le processus plus rapide)
mkdir ../thumbs; for pic in *.* ; do echo "$pic" ; convert -define jpeg:size=220x170 "$pic" -thumbnail 110x85^  -gravity center -extent 110x85  "../thumbs/thumbs_${pic}" ; done

# Créer des thumbnails avec un nom incrémenté
i=0; for pic in *.jpg ;  do echo "$pic"; let i=$i+1 ; fname=`printf "../out/nouveau_nom_%03d.jpg" $i`; convert "$pic" -quality 70 -thumbnail '1000x' $fname; done

# Créer un pdf à partir d’une image jpeg en gardant une taille de fichier similaire
